SELECT
  BRAINSTORMING_SESSION.ID,
  BRAINSTORMING_SESSION.IDEATOR_TYPE,
  BRAINSTORMING_SESSION.IDEATOR_TYPE_SCORE,
  TRACKING_EVENT.TYPE,
  COUNT(TRACKING_EVENT.ID) AS TRACKED_EVENTS
FROM
  BRAINSTORMING_SESSION
  LEFT JOIN TRACKING_EVENT ON BRAINSTORMING_SESSION.ID = TRACKING_EVENT.BRAINSTORMING_SESSION_ID
WHERE
  BRAINSTORMING_SESSION.STATE = 'usable'
  AND BRAINSTORMING_SESSION.CHALLENGE = 'BIONIC_RADAR'
  AND TRACKING_EVENT.TYPE IN ('idea-submit', 'inspiration-request')
GROUP BY
  BRAINSTORMING_SESSION.ID, TRACKING_EVENT.TYPE
ORDER BY
  BRAINSTORMING_SESSION.IDEATOR_TYPE_SCORE
